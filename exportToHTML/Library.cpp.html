72   // method to add an instance of a Subscriber to a library 
73   // param[in]: subscriber (Subscriber) 
74   // param[out]: void 
75   void Library::addSubscriber(Subscriber& subscriber) 
76   { 
77       std::regex pattern(subscriber.getId()); 
78       bool isPresent = true; 
79       if(nSubscribers_ == 0){ 
80           subscribers_[nSubscribers_++] = &subscriber; 
81           std::cout << "Subscriber: #" << subscriber.getId() << " added successfully!\n";
82       } 
83       else{ 
84           isPresent = false; 
85           // Loop to check an existing subscriber in the system 
86           for (int i = 0; i < nSubscribers_; ++i) { 
87               if (regex_match(subscribers_[i]->getId(), pattern)) { 
88                   std::cout << "!!! Subscriber: #" << subscriber.getId() << " failed!!!\n";
89                   std::cout << subscriber.getFirstName() << ", " << subscriber.getLastName()<<". Already exit in the Library!!!\n";
90                   isPresent = true; 
91                   break; 
92               } 
93           } 
94       } 
95    
96       // Add a subscriber if he does not exist in the system 
97       if ((!isPresent) && (nSubscribers_ < MAX_SUB)) { 
98           subscribers_[nSubscribers_++] = &subscriber; 
99           std::cout << "Subscriber: #" << subscriber.getId() << " added successfully!\n";
100      } 
101  }

105  // param[in]: id (string) 
106  // param[out]: void 
107  void Library::removeSubscriber(const std::string &id) 
108  { 
109      if(nSubscribers_ <= 0 || MAX_SUB < nSubscribers_){ 
110          std::logic_error description("Range_Error-- Could Not Remove the Subscriber"); 
111          throw description; 
112      } 
113      else{ 
114          std::regex pattern(id); 
115          for (int i = 0; i < nSubscribers_; i++) { 
116              if (regex_match(subscribers_[i]->getId(), pattern)) { 
117                  for (int j = i; j < nSubscribers_ ; j++) { 
118                      subscribers_[j] = subscribers_[j+1]; 
119                  } 
120                  subscribers_[nSubscribers_ -1] = nullptr; 
121                  nSubscribers_--; 
122                  continue; 
123              } 
124          } 
125      } 
126  }
